14/* Function: config_PWM_A ------------------------------



15 *



16 * Abstract:



17 *      Configure PWM_A



18 */



19void config_PWM_A(uint16_T timerPeriod, uint16_T waveformType,



20                  uint16_T unit1Status, char* unit1Source, uint16_T unit1Value,



21                  uint16_T unit2Status, char* unit2Source, uint16_T unit2Value,



22                  uint16_T unit3Status, char* unit3Source, uint16_T unit3Value,



23                  uint16_T controlLogic,



24                  uint16_T enableDeadband1, uint16_T enableDeadband2, uint16_T



25                  enableDeadband3,



26                  uint16_T deadbandPrescaler, uint16_T deadbandPeriod, uint16_T



27                  timerToADC, uint16_T preScaler)



28{



29  const uint16_T DISABLED= 0;



30



31  /* Initalize EVA Timer1 which controls PWM1-PWM6 */



32  EvaRegs.T1PR = timerPeriod;          /* period*/



33  EvaRegs.T1CNT = 0x0000;              /* counter*/



34  EvaRegs.T1CON.all = 0x1042;          /* enable; compare enable; default TMODE */



35  EvaRegs.T1CON.bit.TMODE = waveformType;/* adjust Timer TMODE*/



36  EvaRegs.T1CON.bit.TPS = preScaler;   /* Input clock prescaler*/



37



38  /* Set ADC SOC signal according to user's request */



39  EvaRegs.GPTCONA.bit.T1TOADC = timerToADC;



40



41  /* Enable compare for PWM1-PWM6 */



42  EvaRegs.CMPR1 = unit1Status ? unit1Value : DISABLED;



43  EvaRegs.CMPR2 = unit2Status ? unit2Value : DISABLED;



44  EvaRegs.CMPR3 = unit3Status ? unit3Value : DISABLED;



45



46  /* Compare action control: Action that takes place on a compare event */



47  EvaRegs.ACTRA.all = controlLogic;



48  EvaRegs.DBTCONA.bit.EDBT1 = enableDeadband1;



49  EvaRegs.DBTCONA.bit.EDBT2 = enableDeadband2;



50  EvaRegs.DBTCONA.bit.EDBT3 = enableDeadband3;



51  EvaRegs.DBTCONA.bit.DBT = deadbandPeriod;



52  EvaRegs.DBTCONA.bit.DBTPS = deadbandPrescaler;



53  EvaRegs.COMCONA.all = 0xA600;



54}

56void config_QEP_A(uint16_T initialCount, uint16_T period, uint16_T qepIndex,



57                  uint16_T qepIndexQual)



58{



59  /* QEP1/QEP2 Control register setting (Timer2)*/



60  EvaRegs.T2PR = period;               /* Timer 2 period register,initial value = 65535        */



61  EvaRegs.T2CNT = initialCount;        /* Timer 2 counter register,initial value = user define*/



62  EvaRegs.T2CON.all = 0x1070;          /* Timer 2 control register: Continuous-Up Count Mode*/



63  EALLOW;



64  GpioMuxRegs.GPAMUX.all |= 0x0300;    /* I/O Mux Control Register GPAMUX: Bit 8~9 Bit as compare*/



65  if (qepIndex == 1) {



66    GpioMuxRegs.GPAMUX.all |= 0x0400;  /* I/O Mux Control Register GPAMUX: Bit 10 Bit as QEP Index*/



67    EvaRegs.EXTCONA.bit.QEPIE = 1;     /* QEP Index Enable*/



68    if (qepIndexQual == 1) {



69      EvaRegs.EXTCONA.bit.QEPIQUAL = 1;/* CAP3_QEPI1 Index Qualification Mode*/



70    }



71  }



72



73  EDIS;



74  EvaRegs.CAPCONA.all |= 0xE000;       /* Enable QEP circuit. Disable Capture Units 1/2 and 3*/



75}

84void init_board ()



85{



86  /* Disable the watchdog        */



87  DisableDog();



88



89  /* Initialize the PLL control: PLLCR



90   * DSP28_PLLCR is calculted from the crystal frequency use input



91   */



92  InitPll(10);



93



94  /* Initialize the peripheral clocks*/



95  InitPeripheralClocks();



96  EALLOW;



97



98  /* Configure low speed peripheral clocks */



99  SysCtrlRegs.LOSPCP.all = 2U;



100



101  /* Configure high speed peripheral clocks */



102  SysCtrlRegs.HISPCP.all = 1U;



103  EDIS;



104



105  /* Perform additional configuration of the XTINF for speed up */



106  XintfRegs.XINTCNF2.bit.XTIMCLK = 0;  /* XTIMCLK=SYSCLKOUT/1*/



107  XintfRegs.XINTCNF2.bit.CLKOFF = 0;   /* XCLKOUT is enabled*/



108  XintfRegs.XINTCNF2.bit.CLKMODE = 0;  /* XCLKOUT = XTIMCLK*/



109



110  /* Make sure write buffer is empty before configuring buffering depth*/



111  while (XintfRegs.XINTCNF2.bit.WLEVEL != 0) ;/* poll the WLEVEL bit*/



112  XintfRegs.XINTCNF2.bit.WRBUFF = 0;   /* No write buffering*/



113



114  /* Example: Assume Zone 7 is slow, so add additional BCYC cycles whenever



115   * switching from Zone 7 to another Zone.  This will help avoid bus contention.



116   */



117  XintfRegs.XBANK.bit.BCYC = 7;        /* Add 7 cycles*/



118  XintfRegs.XBANK.bit.BANK = 7;        /* select zone 7*/



119



120  /* Zone 0 Configuration */



121  /*



122     XintfRegs.XTIMING0.bit.X2TIMING = 0;   // Timing scale factor = 1



123     XintfRegs.XTIMING0.bit.XSIZE = 3;      // Always write as 11b



124     XintfRegs.XTIMING0.bit.READYMODE = 1;  // XREADY is asynchronous



125     XintfRegs.XTIMING0.bit.USEREADY = 0;   // Disable XREADY



126     XintfRegs.XTIMING0.bit.XRDLEAD = 1;    // Read lead time



127     XintfRegs.XTIMING0.bit.XRDACTIVE = 2;  // Read active time



128     XintfRegs.XTIMING0.bit.XRDTRAIL = 0;   // Read trail time



129     XintfRegs.XTIMING0.bit.XWRLEAD = 1;    // Write lead time



130     XintfRegs.XTIMING0.bit.XWRACTIVE = 2;  // Write active time



131     XintfRegs.XTIMING0.bit.XWRTRAIL = 0;   // Write trail time



132   */



133  XintfRegs.XTIMING0.all = (XintfRegs.XTIMING0.all & ~0x43FFFF) | 0x39428;



134



135  /* Zone 1 Configuration */



136  /*



137     XintfRegs.XTIMING1.bit.X2TIMING = 0;    // Timing scale factor = 1



138     XintfRegs.XTIMING1.bit.XSIZE = 3;       // Always write as 11b



139     XintfRegs.XTIMING1.bit.READYMODE = 1;   // XREADY is asynchronous



140     XintfRegs.XTIMING1.bit.USEREADY = 0;    // Disable XREADY



141     XintfRegs.XTIMING1.bit.XRDLEAD = 1;     // Read lead time



142     XintfRegs.XTIMING1.bit.XRDACTIVE = 2;   // Read active time



143     XintfRegs.XTIMING1.bit.XRDTRAIL = 0;    // Read trail time



144     XintfRegs.XTIMING1.bit.XWRLEAD = 1;     // Write lead time



145     XintfRegs.XTIMING1.bit.XWRACTIVE = 2;   // Write active time



146     XintfRegs.XTIMING1.bit.XWRTRAIL = 0;    // Write trail time



147   */



148  XintfRegs.XTIMING1.all = (XintfRegs.XTIMING1.all & ~0x43FFFF) | 0x39428;



149



150  /* Zone 2 Configuration */



151  /*



152     XintfRegs.XTIMING2.bit.X2TIMING = 0;    // Timing scale factor = 1



153     XintfRegs.XTIMING2.bit.XSIZE = 3;       // Always write as 11b



154     XintfRegs.XTIMING2.bit.READYMODE = 1;   // XREADY is asynchronous



155     XintfRegs.XTIMING2.bit.USEREADY = 0;    // Disable XREADY



156     XintfRegs.XTIMING2.bit.XRDLEAD = 1;     // Read lead time



157     XintfRegs.XTIMING2.bit.XRDACTIVE = 2;   // Read active time



158     XintfRegs.XTIMING2.bit.XRDTRAIL = 0;    // Read trail time



159     XintfRegs.XTIMING2.bit.XWRLEAD = 1;     // Write lead time



160     XintfRegs.XTIMING2.bit.XWRACTIVE = 2;   // Write active time



161     XintfRegs.XTIMING2.bit.XWRTRAIL = 0;    // Write trail time



162   */



163  XintfRegs.XTIMING2.all = (XintfRegs.XTIMING2.all & ~0x43FFFF) | 0x39428;



164



165  /* Zone 6 Configuration */



166  /*



167     XintfRegs.XTIMING6.bit.X2TIMING = 0;    // Timing scale factor = 1



168     XintfRegs.XTIMING6.bit.XSIZE = 3;       // Always write as 11b



169     XintfRegs.XTIMING6.bit.READYMODE = 1;   // XREADY is asynchronous



170     XintfRegs.XTIMING6.bit.USEREADY = 0;    // Disable XREADY



171     XintfRegs.XTIMING6.bit.XRDLEAD = 1;     // Read lead time



172     XintfRegs.XTIMING6.bit.XRDACTIVE = 2;   // Read active time



173     XintfRegs.XTIMING6.bit.XRDTRAIL = 0;    // Read trail time



174     XintfRegs.XTIMING6.bit.XWRLEAD = 1;     // Write lead time



175     XintfRegs.XTIMING6.bit.XWRACTIVE = 2;   // Write active time



176     XintfRegs.XTIMING6.bit.XWRTRAIL = 0;    // Write trail time



177   */



178  XintfRegs.XTIMING6.all = (XintfRegs.XTIMING6.all & ~0x43FFFF) | 0x39428;



179



180  /* Zone 7 Configuration */



181  /*



182     XintfRegs.XTIMING7.bit.X2TIMING = 0;    // Timing scale factor = 1



183     XintfRegs.XTIMING7.bit.XSIZE = 3;       // Always write as 11b



184     XintfRegs.XTIMING7.bit.READYMODE = 1;   // XREADY is asynchronous



185     XintfRegs.XTIMING7.bit.USEREADY = 0;    // Disable XREADY



186     XintfRegs.XTIMING7.bit.XRDLEAD = 1;     // Read lead time



187     XintfRegs.XTIMING7.bit.XRDACTIVE = 2;   // Read active time



188     XintfRegs.XTIMING7.bit.XRDTRAIL = 0;    // Read trail time



189     XintfRegs.XTIMING7.bit.XWRLEAD = 1;     // Write lead time



190     XintfRegs.XTIMING7.bit.XWRACTIVE = 2;   // Write active time



191     XintfRegs.XTIMING7.bit.XWRTRAIL = 0;    // Write trail time



192   */



193  XintfRegs.XTIMING7.all = (XintfRegs.XTIMING7.all & ~0x43FFFF) | 0x39428;



194



195  /* Flush pipeline to ensure that the write is complete. Wait to be sure. */



196  asm(" RPT #6 || NOP");



197



198  /* Disable and clear all CPU interrupts */



199  DINT;



200  IER = 0x0000;



201  IFR = 0x0000;



202  InitPieCtrl();



203  InitPieVectTable();



204



205  /* initial eCAN function.... */



206



207  /* initial SPI function.... */



208



209  /* initial SCI function.... */



210  InitCpuTimers();



211



212  /* Initialize GPIO input qualification sampling period */



213  EALLOW;



214  GpioMuxRegs.GPAQUAL.bit.QUALPRD = 0;



215  GpioMuxRegs.GPBQUAL.bit.QUALPRD = 0;



216  GpioMuxRegs.GPDQUAL.bit.QUALPRD = 0;



217  GpioMuxRegs.GPEQUAL.bit.QUALPRD = 0;



218  EDIS;



219}
